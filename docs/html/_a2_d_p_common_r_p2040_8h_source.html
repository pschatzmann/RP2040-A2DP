195 virtual void local_avrcp_controller_packet_handler(uint8_t packet_type,
196 uint16_t channel,
197 uint8_t *packet,
198 uint16_t size) {
199 TRACED();
200 UNUSED(channel);
201 UNUSED(size);
202
203 // helper to print c strings
204 uint8_t avrcp_subevent_value[256];
205 uint8_t play_status;
206
207 // a2dp_sink_arduino_avrcp_connection_t *avrcp_connection =
208 // &a2dp_sink_arduino_avrcp_connection;
209
210 if (packet_type != HCI_EVENT_PACKET)
211 return;
212 if (hci_event_packet_get_type(packet) != HCI_EVENT_AVRCP_META)
213 return;
214 if (get_avrcp_cid() == 0)
215 return;
216
217 memset(avrcp_subevent_value, 0, sizeof(avrcp_subevent_value));
218 switch (packet[2]) {
219
220 case AVRCP_SUBEVENT_NOTIFICATION_VOLUME_CHANGED: {
221 int vol = avrcp_subevent_notification_volume_changed_get_absolute_volume(
222 packet) *
223 100 / 127;
224 LOGI("AVRCP Controller: Notification Absolute Volume %d %%", vol);
225 avrcp_volume_changed(vol);
226 } break;
227 case AVRCP_SUBEVENT_NOTIFICATION_EVENT_BATT_STATUS_CHANGED:
228 // see avrcp_battery_status_t
229 LOGI(
230 "AVRCP Controller: Notification Battery Status %d",
231 avrcp_subevent_notification_event_batt_status_changed_get_battery_status(
232 packet));
233 break;
234 case AVRCP_SUBEVENT_NOTIFICATION_STATE:
235 LOGI("AVRCP Controller: Notification %s - %s",
236 avrcp_event2str(
237 avrcp_subevent_notification_state_get_event_id(packet)),
238 avrcp_subevent_notification_state_get_enabled(packet) != 0
239 ? "enabled"
240 : "disabled");
241 break;
242
243 case AVRCP_SUBEVENT_NOTIFICATION_PLAYBACK_POS_CHANGED: {
244 uint32_t pos_ms =
245 avrcp_subevent_notification_playback_pos_changed_get_playback_position_ms(
246 packet);
247 LOGI("AVRCP Controller: Playback position changed, position %d ms",
248 (unsigned int)pos_ms);
249 if (metadata_callback)
250 metadata_callback(MDPlaybackPosMs, nullptr, pos_ms);
251 } break;
252 case AVRCP_SUBEVENT_NOTIFICATION_PLAYBACK_STATUS_CHANGED: {
253 play_status =
254 avrcp_subevent_notification_playback_status_changed_get_play_status(
255 packet);
256 if (is_valid_playstatus(play_status)) {
257 LOGI("AVRCP Controller: Playback status changed %s",
258 avrcp_play_status2str(play_status));
259 }
260 switch (play_status) {
261 case AVRCP_PLAYBACK_STATUS_PLAYING:
262 // avrcp_connection->playing = true;
263 set_playing(true);
264 break;
265 default:
266 // avrcp_connection->playing = false;
267 set_playing(false);
268 break;
269 }
270
271 if (is_valid_playstatus(play_status)) {
272 LOGI("AVRCP Controller: Playback status changed %s",
273 avrcp_play_status2str(play_status));
274 }
275 }
276 return;
277 case AVRCP_SUBEVENT_NOTIFICATION_NOW_PLAYING_CONTENT_CHANGED:
278 LOGI("AVRCP Controller: Playing content changed");
279 return;
280 case AVRCP_SUBEVENT_NOTIFICATION_TRACK_CHANGED:
281 LOGI("AVRCP Controller: Track changed");
282 return;
283 case AVRCP_SUBEVENT_NOTIFICATION_AVAILABLE_PLAYERS_CHANGED:
284 LOGI("AVRCP Controller: Changed");
285 return;
286 case AVRCP_SUBEVENT_SHUFFLE_AND_REPEAT_MODE: {
287 uint8_t shuffle_mode =
288 avrcp_subevent_shuffle_and_repeat_mode_get_shuffle_mode(packet);
289 uint8_t repeat_mode =
290 avrcp_subevent_shuffle_and_repeat_mode_get_repeat_mode(packet);
291 LOGI("AVRCP Controller: %s, %s", avrcp_shuffle2str(shuffle_mode),
292 avrcp_repeat2str(repeat_mode));
293 break;
294 }
295 case AVRCP_SUBEVENT_NOW_PLAYING_TRACK_INFO: {
296 uint32_t val = avrcp_subevent_now_playing_track_info_get_track(packet);
297 LOGI("AVRCP Controller: Track: %d", val);
298 if (metadata_callback)
299 metadata_callback(MDTrack, nullptr, val);
300 } break;
301
302 case AVRCP_SUBEVENT_NOW_PLAYING_TOTAL_TRACKS_INFO: {
303 uint32_t val =
304 avrcp_subevent_now_playing_total_tracks_info_get_total_tracks(packet);
305 LOGI("AVRCP Controller: Total Tracks: %d", val);
306 if (metadata_callback)
307 metadata_callback(MDTracks, nullptr, val);
308 } break;
309
310 case AVRCP_SUBEVENT_NOW_PLAYING_TITLE_INFO:
311 if (avrcp_subevent_now_playing_title_info_get_value_len(packet) > 0) {
312 memcpy(avrcp_subevent_value,
313 avrcp_subevent_now_playing_title_info_get_value(packet),
314 avrcp_subevent_now_playing_title_info_get_value_len(packet));
315 LOGI("AVRCP Controller: Title: %s", avrcp_subevent_value);
316 if (metadata_callback)
317 metadata_callback(MDTitle, (const char *)avrcp_subevent_value, 0);
318 }
319 break;
320
321 case AVRCP_SUBEVENT_NOW_PLAYING_ARTIST_INFO:
322 if (avrcp_subevent_now_playing_artist_info_get_value_len(packet) > 0) {
323 memcpy(avrcp_subevent_value,
324 avrcp_subevent_now_playing_artist_info_get_value(packet),
325 avrcp_subevent_now_playing_artist_info_get_value_len(packet));
326 LOGI("AVRCP Controller: Artist: %s", avrcp_subevent_value);
327 if (metadata_callback)
328 metadata_callback(MDArtist, (const char *)avrcp_subevent_value, 0);
329 }
330 break;
331
332 case AVRCP_SUBEVENT_NOW_PLAYING_ALBUM_INFO:
333 if (avrcp_subevent_now_playing_album_info_get_value_len(packet) > 0) {
334 memcpy(avrcp_subevent_value,
335 avrcp_subevent_now_playing_album_info_get_value(packet),
336 avrcp_subevent_now_playing_album_info_get_value_len(packet));
337 LOGI("AVRCP Controller: Album: %s", avrcp_subevent_value);
338 if (metadata_callback)
339 metadata_callback(MDAlbum, (const char *)avrcp_subevent_value, 0);
340 }
341 break;
342
343 case AVRCP_SUBEVENT_NOW_PLAYING_GENRE_INFO:
344 if (avrcp_subevent_now_playing_genre_info_get_value_len(packet) > 0) {
345 memcpy(avrcp_subevent_value,
346 avrcp_subevent_now_playing_genre_info_get_value(packet),
347 avrcp_subevent_now_playing_genre_info_get_value_len(packet));
348 LOGI("AVRCP Controller: Genre: %s", avrcp_subevent_value);
349 if (metadata_callback)
350 metadata_callback(MDGenre, (const char *)avrcp_subevent_value, 0);
351 }
352 break;
353
354 case AVRCP_SUBEVENT_PLAY_STATUS: {
355 uint32_t len = avrcp_subevent_play_status_get_song_length(packet);
356 uint32_t pos = avrcp_subevent_play_status_get_song_position(packet);
357 LOGI("AVRCP Controller: Song length %" PRIu32
358 " ms, Song position %" PRIu32 " ms, Play status %s",
359 len, pos,
360 avrcp_play_status2str(
361 avrcp_subevent_play_status_get_play_status(packet)));
362
363 if (metadata_callback) {
364 metadata_callback(MDSongLen, nullptr, len);
365 metadata_callback(MDSongPos, nullptr, pos);
366 }
367 } break;
368
369 case AVRCP_SUBEVENT_OPERATION_COMPLETE:
370 LOGI("AVRCP Controller: %s complete",
371 avrcp_operation2str(
372 avrcp_subevent_operation_complete_get_operation_id(packet)));
373 break;
374
375 case AVRCP_SUBEVENT_OPERATION_START:
376 LOGI("AVRCP Controller: %s start",
377 avrcp_operation2str(
378 avrcp_subevent_operation_start_get_operation_id(packet)));
379 break;
380
381 case AVRCP_SUBEVENT_NOTIFICATION_EVENT_TRACK_REACHED_END:
382 LOGI("AVRCP Controller: Track reached end");
383 break;
384
385 case AVRCP_SUBEVENT_PLAYER_APPLICATION_VALUE_RESPONSE:
386 LOGI(
387 "AVRCP Controller: Set Player App Value %s",
388 avrcp_ctype2str(
389 avrcp_subevent_player_application_value_response_get_command_type(
390 packet)));
391 break;
392
393 default:
394 break;
395 }
396 }

398 virtual int16_t get_max_input_amplitude() = 0;
399
400 void avrcp_volume_changed(uint8_t volume) {
401 // calculate eff volume factor
402 float max_current = get_max_input_amplitude();
403 float max_factor = 32767.0f / max_current;
404 float vol_float = 0.01f * volume_percentage;
405 // We adjust the volume: if we receive only max amplitude of 2500 we can
406 // multiply the values by a factor of 13
407 float factor = mapFloat(vol_float, 0.0, 1.0, 0.0, max_factor);
408 LOGI("avrcp_volume_changed: %d -> %f", volume, factor);
409 // adjust volume
410 volume_stream.setVolume(factor);
411 }
