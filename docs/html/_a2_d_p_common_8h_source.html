231 
232  // a2dp_sink_arduino_avrcp_connection_t *avrcp_connection =
233  // &a2dp_sink_arduino_avrcp_connection;
234 
235  if (packet_type != HCI_EVENT_PACKET)
236  return;
237  if (hci_event_packet_get_type(packet) != HCI_EVENT_AVRCP_META)
238  return;
239  if (get_avrcp_cid() == 0)
240  return;
241 
242  memset(avrcp_subevent_value, 0, sizeof(avrcp_subevent_value));
243  switch (packet[2]) {
244 
245  case AVRCP_SUBEVENT_NOTIFICATION_VOLUME_CHANGED: {
246  int vol = volume_to_percent(avrcp_subevent_notification_volume_changed_get_absolute_volume(
247  packet));
248  LOGI("AVRCP Controller: Notification Absolute Volume %d %%", vol);
249  avrcp_volume_changed(vol);
250  } break;
251  case AVRCP_SUBEVENT_NOTIFICATION_EVENT_BATT_STATUS_CHANGED:
252  // see avrcp_battery_status_t
253  LOGI(
254  "AVRCP Controller: Notification Battery Status %d",
255  avrcp_subevent_notification_event_batt_status_changed_get_battery_status(
256  packet));
257  break;
258  case AVRCP_SUBEVENT_NOTIFICATION_STATE:
259  LOGI("AVRCP Controller: Notification %s - %s",
260  avrcp_event2str(
261  avrcp_subevent_notification_state_get_event_id(packet)),
262  avrcp_subevent_notification_state_get_enabled(packet) != 0
263  ? "enabled"
264  : "disabled");
265  break;
266 
267  case AVRCP_SUBEVENT_NOTIFICATION_PLAYBACK_POS_CHANGED: {
268  uint32_t pos_ms =
269  avrcp_subevent_notification_playback_pos_changed_get_playback_position_ms(
270  packet);
271  LOGI("AVRCP Controller: Playback position changed, position %d ms",
272  (unsigned int)pos_ms);
273  if (metadata_callback)
274  metadata_callback(MDPlaybackPosMs, nullptr, pos_ms);
275  } break;
276  case AVRCP_SUBEVENT_NOTIFICATION_PLAYBACK_STATUS_CHANGED: {
277  play_status =
278  avrcp_subevent_notification_playback_status_changed_get_play_status(
279  packet);
280  if (is_valid_playstatus(play_status)) {
281  LOGI("AVRCP Controller: Playback status changed %s",
282  avrcp_play_status2str(play_status));
283  }
284  switch (play_status) {
285  case AVRCP_PLAYBACK_STATUS_PLAYING:
286  // avrcp_connection->playing = true;
287  set_playing(true);
288  break;
289  default:
290  // avrcp_connection->playing = false;
291  set_playing(false);
292  break;
293  }
294 
295  if (is_valid_playstatus(play_status)) {
296  LOGI("AVRCP Controller: Playback status changed %s",
297  avrcp_play_status2str(play_status));
298  }
299  }
300  return;
301  case AVRCP_SUBEVENT_NOTIFICATION_NOW_PLAYING_CONTENT_CHANGED:
302  LOGI("AVRCP Controller: Playing content changed");
303  return;
304  case AVRCP_SUBEVENT_NOTIFICATION_TRACK_CHANGED:
305  LOGI("AVRCP Controller: Track changed");
306  return;
307  case AVRCP_SUBEVENT_NOTIFICATION_AVAILABLE_PLAYERS_CHANGED:
308  LOGI("AVRCP Controller: Changed");
309  return;
310  case AVRCP_SUBEVENT_SHUFFLE_AND_REPEAT_MODE: {
311  uint8_t shuffle_mode =
312  avrcp_subevent_shuffle_and_repeat_mode_get_shuffle_mode(packet);
313  uint8_t repeat_mode =
314  avrcp_subevent_shuffle_and_repeat_mode_get_repeat_mode(packet);
315  LOGI("AVRCP Controller: %s, %s", avrcp_shuffle2str(shuffle_mode),
316  avrcp_repeat2str(repeat_mode));
317  break;
318  }
319  case AVRCP_SUBEVENT_NOW_PLAYING_TRACK_INFO: {
320  uint32_t val = avrcp_subevent_now_playing_track_info_get_track(packet);
321  LOGI("AVRCP Controller: Track: %d", val);
322  if (metadata_callback)
323  metadata_callback(MDTrack, nullptr, val);
324  } break;
325 
326  case AVRCP_SUBEVENT_NOW_PLAYING_TOTAL_TRACKS_INFO: {
327  uint32_t val =
328  avrcp_subevent_now_playing_total_tracks_info_get_total_tracks(packet);
329  LOGI("AVRCP Controller: Total Tracks: %d", val);
330  if (metadata_callback)
331  metadata_callback(MDTracks, nullptr, val);
332  } break;
333 
334  case AVRCP_SUBEVENT_NOW_PLAYING_TITLE_INFO:
335  if (avrcp_subevent_now_playing_title_info_get_value_len(packet) > 0) {
336  memcpy(avrcp_subevent_value,
337  avrcp_subevent_now_playing_title_info_get_value(packet),
338  avrcp_subevent_now_playing_title_info_get_value_len(packet));
339  LOGI("AVRCP Controller: Title: %s", avrcp_subevent_value);
340  if (metadata_callback)
341  metadata_callback(MDTitle, (const char *)avrcp_subevent_value, 0);
342  }
343  break;
344 
345  case AVRCP_SUBEVENT_NOW_PLAYING_ARTIST_INFO:
346  if (avrcp_subevent_now_playing_artist_info_get_value_len(packet) > 0) {
347  memcpy(avrcp_subevent_value,
348  avrcp_subevent_now_playing_artist_info_get_value(packet),
349  avrcp_subevent_now_playing_artist_info_get_value_len(packet));
350  LOGI("AVRCP Controller: Artist: %s", avrcp_subevent_value);
351  if (metadata_callback)
352  metadata_callback(MDArtist, (const char *)avrcp_subevent_value, 0);
353  }
354  break;
355 
356  case AVRCP_SUBEVENT_NOW_PLAYING_ALBUM_INFO:
357  if (avrcp_subevent_now_playing_album_info_get_value_len(packet) > 0) {
358  memcpy(avrcp_subevent_value,
359  avrcp_subevent_now_playing_album_info_get_value(packet),
360  avrcp_subevent_now_playing_album_info_get_value_len(packet));
361  LOGI("AVRCP Controller: Album: %s", avrcp_subevent_value);
362  if (metadata_callback)
363  metadata_callback(MDAlbum, (const char *)avrcp_subevent_value, 0);
364  }
365  break;
366 
367  case AVRCP_SUBEVENT_NOW_PLAYING_GENRE_INFO:
368  if (avrcp_subevent_now_playing_genre_info_get_value_len(packet) > 0) {
369  memcpy(avrcp_subevent_value,
370  avrcp_subevent_now_playing_genre_info_get_value(packet),
371  avrcp_subevent_now_playing_genre_info_get_value_len(packet));
372  LOGI("AVRCP Controller: Genre: %s", avrcp_subevent_value);
373  if (metadata_callback)
374  metadata_callback(MDGenre, (const char *)avrcp_subevent_value, 0);
375  }
376  break;
377 
378  case AVRCP_SUBEVENT_PLAY_STATUS: {
379  uint32_t len = avrcp_subevent_play_status_get_song_length(packet);
380  uint32_t pos = avrcp_subevent_play_status_get_song_position(packet);
381  LOGI("AVRCP Controller: Song length %" PRIu32
382  " ms, Song position %" PRIu32 " ms, Play status %s",
383  len, pos,
384  avrcp_play_status2str(
385  avrcp_subevent_play_status_get_play_status(packet)));
386 
387  if (metadata_callback) {
388  metadata_callback(MDSongLen, nullptr, len);
389  metadata_callback(MDSongPos, nullptr, pos);
390  }
391  } break;
392 
393  case AVRCP_SUBEVENT_OPERATION_COMPLETE:
394  LOGI("AVRCP Controller: %s complete",
395  avrcp_operation2str(
396  avrcp_subevent_operation_complete_get_operation_id(packet)));
397  break;
398 
399  case AVRCP_SUBEVENT_OPERATION_START:
400  LOGI("AVRCP Controller: %s start",
401  avrcp_operation2str(
402  avrcp_subevent_operation_start_get_operation_id(packet)));
403  break;
404 
405  case AVRCP_SUBEVENT_NOTIFICATION_EVENT_TRACK_REACHED_END:
406  LOGI("AVRCP Controller: Track reached end");
407  break;
408 
409  case AVRCP_SUBEVENT_PLAYER_APPLICATION_VALUE_RESPONSE:
410  LOGI(
411  "AVRCP Controller: Set Player App Value %s",
412  avrcp_ctype2str(
413  avrcp_subevent_player_application_value_response_get_command_type(
414  packet)));
415  break;
416 
417  default:
418  break;

424  // volume is from 0 to 100
425  void avrcp_volume_changed(uint8_t volume) {
426  // calculate eff volume factor
427  float max_current = get_max_input_amplitude();
428  float max_factor = 32767.0f / max_current;
429  float vol_float = volume_as_float(volume_percentage);
430  // We adjust the volume: if we receive only max amplitude of 2500 we can
431  // multiply the values by a factor of 13
432  float factor = mapFloat(vol_float, 0.0, 1.0, 0.0, max_factor);
433  LOGI("avrcp_volume_changed: %d -> %f", volume, factor);
434  // adjust volume
435  volume_stream.setVolume(factor);
436  }
