41 /* EXAMPLE_START(a2dp_source_arduino): A2DP Source - Stream Audio and Control
42  * Volume
43  *
44  * @text This A2DP Source example arduinonstrates how to send an audio data
45  * stream to a remote A2DP Sink device and how to switch between two audio data
46  * sources. In addition, the AVRCP Target is used to answer queries on currently
47  * played media, as well as to handle remote playback control, i.e. play, stop,
48  * repeat, etc.
49  *
50  * @text To test with a remote device, e.g. a Bluetooth speaker,
51  * set the device_addr_string to the Bluetooth address of your
52  * remote device in the code, and use the UI to connect and start playback.
53  *
54  * @text For more info on BTstack audio, see our blog post
55  * [A2DP Sink and Source on STM32 F4 Discovery
56  * Board](http://bluekitchen-gmbh.com/a2dp-sink-and-source-on-stm32-f4-discovery-board/).

207  int a2dp_source_and_avrcp_services_init(void) {
208  TRACED();
209 
210  // Request role change on reconnecting headset to always use them in slave
211  // mode
212  hci_set_master_slave_policy(0);
213  // enabled EIR
214  hci_set_inquiry_mode(INQUIRY_MODE_RSSI_AND_EIR);
215 
216  l2cap_init();
217 
218  if (isBLEEnabled()) {
219  // Initialize LE Security Manager. Needed for cross-transport key
220  // derivation
221  sm_init();
222  }
223 
224  // Initialize A2DP Source
225  a2dp_source_init();
226  a2dp_source_register_packet_handler(&source_a2dp_packet_handler);
227 
228  // Create stream endpoint
229  A2DPEncoder &enc = get_encoder();
230  avdtp_stream_endpoint_t *local_stream_endpoint =
231  a2dp_source_create_stream_endpoint(
232  AVDTP_AUDIO, enc.codecType(), enc.codecCapabilities(),
233  enc.codecCapabilitiesSize(), enc.config(), enc.configSize());
234  if (!local_stream_endpoint) {
235  LOGE("A2DP Source: not enough memory to create local stream endpoint");
236  return 1;
237  }
238 
239  // Store stream enpoint's SEP ID, as it is used by A2DP API to indentify the
240  // stream endpoint
241  media_tracker.local_seid = avdtp_local_seid(local_stream_endpoint);
242  avdtp_source_register_delay_reporting_category(media_tracker.local_seid);
243 
244  // Initialize AVRCP Service
245  avrcp_init();
246  avrcp_register_packet_handler(&source_avrcp_packet_handler);
247  // Initialize AVRCP Target
248  avrcp_target_init();
249  avrcp_target_register_packet_handler(&source_avrcp_target_packet_handler);
250 
251  // Initialize AVRCP Controller
252  avrcp_controller_init();
253  avrcp_controller_register_packet_handler(
254  &source_avrcp_controller_packet_handler);
255 
256  // Initialize SDP,
257  sdp_init();
258 
259  // Create A2DP Source service record and register it with SDP
260  memset(sdp_a2dp_source_service_buffer, 0,
261  sizeof(sdp_a2dp_source_service_buffer));
262  a2dp_source_create_sdp_record(sdp_a2dp_source_service_buffer, 0x10001,
263  AVDTP_SOURCE_FEATURE_MASK_PLAYER, NULL, NULL);
264  sdp_register_service(sdp_a2dp_source_service_buffer);
265 
266  // Create AVRCP Target service record and register it with SDP. We receive
267  // Category 1 commands from the headphone, e.g. play/pause
268  memset(sdp_avrcp_target_service_buffer, 0,
269  sizeof(sdp_avrcp_target_service_buffer));
270  uint16_t supported_features =
271  AVRCP_FEATURE_MASK_CATEGORY_PLAYER_OR_RECORDER;
272 #ifdef AVRCP_BROWSING_ENABLED
273  supported_features |= AVRCP_FEATURE_MASK_BROWSING;
274 #endif
275  avrcp_target_create_sdp_record(sdp_avrcp_target_service_buffer, 0x10002,
276  supported_features, NULL, NULL);
277  sdp_register_service(sdp_avrcp_target_service_buffer);
278 
279  // Create AVRCP Controller service record and register it with SDP. We send
280  // Category 2 commands to the headphone, e.g. volume up/down
281  memset(sdp_avrcp_controller_service_buffer, 0,
282  sizeof(sdp_avrcp_controller_service_buffer));
283  uint16_t controller_supported_features =
284  AVRCP_FEATURE_MASK_CATEGORY_MONITOR_OR_AMPLIFIER;
285  avrcp_controller_create_sdp_record(sdp_avrcp_controller_service_buffer,
286  0x10003, controller_supported_features,
287  NULL, NULL);
288  sdp_register_service(sdp_avrcp_controller_service_buffer);
289 
290  // Register Device ID (PnP) service SDP record
291  memset(device_id_sdp_service_buffer, 0,
292  sizeof(device_id_sdp_service_buffer));
293  device_id_create_sdp_record(device_id_sdp_service_buffer, 0x10004,
294  DEVICE_ID_VENDOR_ID_SOURCE_BLUETOOTH,
295  BLUETOOTH_COMPANY_ID_BLUEKITCHEN_GMBH, 1, 1);
296  sdp_register_service(device_id_sdp_service_buffer);
297 
298  // Set local name with a template Bluetooth address, that will be
299  // automatically replaced with a actual address once it is available, i.e.
300  // when BTstack boots up and starts talking to a Bluetooth module.
301  gap_set_local_name("A2DP Source 00:00:00:00:00:00");
302  gap_discoverable_control(1);
303  gap_set_class_of_device(0x200408);
304 
305  // Register for HCI events.
306  hci_event_callback_registration.callback = &source_hci_packet_handler;
307  hci_add_event_handler(&hci_event_callback_registration);
308 
309  source_a2dp_configure_sample_rate(current_sample_rate);
310 
311  // Parse human readable Bluetooth address.
312  sscanf_bd_addr(device_addr_string, device_addr);
313 
314  return 0;
315  }

364  void a2dp_arduino_send_media_packet(void) {
365  TRACED();
366 
367  // determine data
368  int num_bytes_in_frame = sbc_buffer_length_sbc();
369  int available = media_tracker.queue.available();
370  int num_frames = available / num_bytes_in_frame;
371 
372  // log output
373  LOGI("a2dp_arduino_send_media_packet: %d packets (%d bytes)", num_frames,
374  available);
375  if (available % num_bytes_in_frame) {
376  LOGW("Invalid number of available bytes: available: %d, frame_size: %d",
377  available, num_bytes_in_frame);
378  }
379 
380  // send out data
381  uint8_t buffer[available + 1];
382  // Prepend SBC Header // (fragmentation << 7) | (starting_packet << 6) |
383  // (last_packet << 5) | num_frames;
384  media_tracker.queue.readBytes(buffer + 1, available);
385  buffer[0] = num_frames;
386  int rc = avdtp_source_stream_send_media_payload_rtp(
387  media_tracker.a2dp_cid, media_tracker.local_seid, 0, buffer,
388  available + 1);
389 
390  if (rc != ERROR_CODE_SUCCESS) {
391  LOGE("avdtp_source_stream_send_media_payload_rtp: %d", rc);
392  }
393 
394  // allow to process the next packets
395  media_tracker.sbc_is_busy = false;
396  }

398  int a2dp_arduino_fill_sbc_audio_buffer(
399  a2dp_media_sending_context_t *context) {
400  TRACED();
401  int len = sbc_buffer_length_pcm() * SBC_PACKET_COUNT;
402  uint8_t pcm_buffer[len];
403  while (media_tracker.queue.available() == 0) {
404  size_t bytes = volume_stream.readBytes(pcm_buffer, len);
405  LOGD("readBytes: %d -> %d", len, bytes);
406 
407  size_t bytes_written = encoder_stream.write(pcm_buffer, bytes);
408  LOGD("write: %d -> %d", bytes_written, media_tracker.queue.available());
409  }
410  int available = media_tracker.queue.available();
411  LOGD("sbc bytes: %d", available);
412  return available;
413  }

467  void local_hci_packet_handler(uint8_t packet_type, uint16_t channel,
468  uint8_t *packet, uint16_t size) {
469  TRACED();
470  UNUSED(channel);
471  UNUSED(size);
472  if (packet_type != HCI_EVENT_PACKET) return;
473  uint8_t status;
474  UNUSED(status);
475 
476  bd_addr_t address;
477  uint32_t cod;
478  bool name_matches = remote_name == nullptr;
479 
480  // Service Class: Rendering | Audio, Major Device Class: Audio
481  const uint32_t bluetooth_speaker_cod = 0x200000 | 0x040000 | 0x000400;
482 
483  switch (hci_event_packet_get_type(packet)) {
484  case BTSTACK_EVENT_STATE:
485  if (btstack_event_state_get_state(packet) != HCI_STATE_WORKING) return;
486  a2dp_source_arduino_start_scanning();
487  break;
488  case HCI_EVENT_PIN_CODE_REQUEST:
489  LOGI("Pin code request - using '0000'");
490  hci_event_pin_code_request_get_bd_addr(packet, address);
491  gap_pin_code_response(address, "0000");
492  break;
493  case GAP_EVENT_INQUIRY_RESULT:
494  gap_event_inquiry_result_get_bd_addr(packet, address);
495  // print info
496  LOGI("Device found: %s ", bd_addr_to_str(address));
497  cod = gap_event_inquiry_result_get_class_of_device(packet);
498  LOGI("- COD: %06" PRIx32, cod);
499  if (gap_event_inquiry_result_get_rssi_available(packet)) {
500  LOGI("- rssi %d dBm",
501  (int8_t)gap_event_inquiry_result_get_rssi(packet));
502  }
503  if (gap_event_inquiry_result_get_name_available(packet)) {
504  char name_buffer[240];
505  int name_len = gap_event_inquiry_result_get_name_len(packet);
506  memcpy(name_buffer, gap_event_inquiry_result_get_name(packet),
507  name_len);
508  name_buffer[name_len] = 0;
509  LOGI("- name '%s'", name_buffer);
510  // check if the name is the requested remote name
511  if (remote_name != nullptr) {
512  name_matches = Str(name_buffer).equalsIgnoreCase(remote_name);
513  }
514  }
515  if (name_matches &&
516  (cod & bluetooth_speaker_cod) == bluetooth_speaker_cod) {
517  memcpy(device_addr, address, 6);
518  LOGI("Bluetooth speaker detected, trying to connect to %s...",
519  bd_addr_to_str(device_addr));
520  scan_active = false;
521  gap_inquiry_stop();
522  a2dp_source_establish_stream(device_addr, &media_tracker.a2dp_cid);
523  }
524  break;
525  case GAP_EVENT_INQUIRY_COMPLETE:
526  if (scan_active) {
527  LOGW("No Bluetooth speakers found, scanning again...");
528  gap_inquiry_start(A2DP_SOURCE_arduino_INQUIRY_DURATION_1280MS);
529  }
530  break;
531  default:
532  break;
533  }
534  }

536  void local_a2dp_packet_handler(uint8_t packet_type, uint16_t channel,
537  uint8_t *packet, uint16_t size) {
538  TRACED();
539  UNUSED(channel);
540  UNUSED(size);
541  uint8_t status;
542  uint8_t local_seid;
543  bd_addr_t address;
544  uint16_t cid;
545 
546  avdtp_channel_mode_t channel_mode;
547  uint8_t allocation_method;
548 
549  if (packet_type != HCI_EVENT_PACKET) return;
550  if (hci_event_packet_get_type(packet) != HCI_EVENT_A2DP_META) return;
551 
552  switch (hci_event_a2dp_meta_get_subevent_code(packet)) {
553  case A2DP_SUBEVENT_SIGNALING_CONNECTION_ESTABLISHED:
554  a2dp_subevent_signaling_connection_established_get_bd_addr(packet,
555  address);
556  cid =
557  a2dp_subevent_signaling_connection_established_get_a2dp_cid(packet);
558  status =
559  a2dp_subevent_signaling_connection_established_get_status(packet);
560 
561  if (status != ERROR_CODE_SUCCESS) {
562  LOGE(
563  "A2DP Source: Connection failed, status 0x%02x, cid 0x%02x, "
564  "a2dp_cid 0x%02x",
565  status, cid, media_tracker.a2dp_cid);
566  media_tracker.a2dp_cid = 0;
567  break;
568  }
569  media_tracker.a2dp_cid = cid;
570  LOGI(
571  "A2DP Source: Connected to address %s, a2dp cid 0x%02x, local "
572  "seid 0x%02x.",
573  bd_addr_to_str(address), media_tracker.a2dp_cid,
574  media_tracker.local_seid);
575  break;
576 
577  case A2DP_SUBEVENT_SIGNALING_MEDIA_CODEC_SBC_CONFIGURATION: {
578  cid =
579  avdtp_subevent_signaling_media_codec_sbc_configuration_get_avdtp_cid(
580  packet);
581  if (cid != media_tracker.a2dp_cid) return;
582 
583  media_tracker.remote_seid =
584  a2dp_subevent_signaling_media_codec_sbc_configuration_get_remote_seid(
585  packet);
586 
587  A2DPEncoder &enc = get_encoder();
588  enc.setValues(cid, packet, size);
589  // Setup SBC decoder
590  auto info = enc.audioInfo();
591  source_a2dp_configure_sample_rate(info.sample_rate);
592  open_audio_streams();
593  break;
594  }
595 
596  case A2DP_SUBEVENT_SIGNALING_DELAY_REPORTING_CAPABILITY:
597  LOGI(
598  "A2DP Source: remote supports delay report, remote seid %d",
599  avdtp_subevent_signaling_delay_reporting_capability_get_remote_seid(
600  packet));

742  void local_avrcp_packet_handler(uint8_t packet_type, uint16_t channel,
743  uint8_t *packet, uint16_t size) {
744  TRACED();
745  UNUSED(channel);
746  UNUSED(size);
747  bd_addr_t event_addr;
748  uint16_t local_cid;
749  uint8_t status = ERROR_CODE_SUCCESS;
750 
751  if (packet_type != HCI_EVENT_PACKET) return;
752  if (hci_event_packet_get_type(packet) != HCI_EVENT_AVRCP_META) return;
753 
754  switch (packet[2]) {
755  case AVRCP_SUBEVENT_CONNECTION_ESTABLISHED:
756  local_cid = avrcp_subevent_connection_established_get_avrcp_cid(packet);
757  status = avrcp_subevent_connection_established_get_status(packet);
758  if (status != ERROR_CODE_SUCCESS) {
759  LOGE("AVRCP: Connection failed, local cid 0x%02x, status 0x%02x",
760  local_cid, status);
761  return;
762  }
763  media_tracker.avrcp_cid = local_cid;
764  avrcp_subevent_connection_established_get_bd_addr(packet, event_addr);
765 
766  LOGI("AVRCP: Channel to %s successfully opened, avrcp_cid 0x%02x",
767  bd_addr_to_str(event_addr), media_tracker.avrcp_cid);
768 
769  avrcp_target_support_event(
770  media_tracker.avrcp_cid,
771  AVRCP_NOTIFICATION_EVENT_PLAYBACK_STATUS_CHANGED);
772  avrcp_target_support_event(media_tracker.avrcp_cid,
773  AVRCP_NOTIFICATION_EVENT_TRACK_CHANGED);
774  avrcp_target_support_event(
775  media_tracker.avrcp_cid,
776  AVRCP_NOTIFICATION_EVENT_NOW_PLAYING_CONTENT_CHANGED);
777  avrcp_target_set_now_playing_info(media_tracker.avrcp_cid, NULL,
778  track_count);
779 
780  LOGI("Enable Volume Change notification");
781  avrcp_controller_enable_notification(
782  media_tracker.avrcp_cid, AVRCP_NOTIFICATION_EVENT_VOLUME_CHANGED);
783  LOGI("Enable Battery Status Change notification");
784  avrcp_controller_enable_notification(
785  media_tracker.avrcp_cid,
786  AVRCP_NOTIFICATION_EVENT_BATT_STATUS_CHANGED);
787  return;
788 
789  case AVRCP_SUBEVENT_CONNECTION_RELEASED:
790  LOGI("AVRCP Target: Disconnected, avrcp_cid 0x%02x",
791  avrcp_subevent_connection_released_get_avrcp_cid(packet));
792  media_tracker.avrcp_cid = 0;
793  return;
794  default:
795  break;
796  }
797 
798  if (status != ERROR_CODE_SUCCESS) {
799  LOGE("Responding to event 0x%02x failed with status 0x%02x", packet[2],
800  status);
801  }

804  void local_avrcp_target_packet_handler(uint8_t packet_type, uint16_t channel,
805  uint8_t *packet, uint16_t size) {
